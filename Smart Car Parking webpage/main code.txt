        document.getElementById('occupied-slots').querySelector('h3').innerText = state.occupiedSlots;

     function bookSlot() {
        if (state.occupiedSlots < state.totalSlots && state.users > 0) {
            state.occupiedSlots++;
            state.users--;
            updateDashboard();
            alert('Slot booked successfully!');
        } else if (state.occupiedSlots >= state.totalSlots) {
            alert('No available slots!');
        } else {
            alert('No users available to book a slot!');
        }
    }

function calculateFee() {
    const slotType = document.getElementById('slot-type').value;
    const bookingTime = document.getElementById('booking-time').value;
    let baseFee = slotRates[slotType];

    if (bookingTime) {
        const [hours, minutes] = bookingTime.split(':').map(Number);
        if ((hours >= 18 && hours <= 23) || (hours >= 0 && hours < 6)) {
            baseFee += 5;  // Additional charge for night time
        }
    }

    document.getElementById('fee').value = baseFee;
}

document.getElementById('book-button').addEventListener('click', function() {
    var slotsFree = document.querySelector('.slots-free');
    var slotsOccupied = document.querySelector('.slots-occupied');
    
    // Decrease slots free and increase occupied slots
    var slotsFreeCount = parseInt(slotsFree.textContent);
    var slotsOccupiedCount = parseInt(slotsOccupied.textContent);
    if (slotsFreeCount > 0) {
        slotsFree.textContent = slotsFreeCount - 1;
        slotsOccupied.textContent = slotsOccupiedCount + 1;
        alert('Slot booked');
    } else {
        alert('No slots available');
    }
});
